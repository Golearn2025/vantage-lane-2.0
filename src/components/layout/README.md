```tsx

  return (
    <Layout
      hideNavbar
      hideFooter
      fullHeight
      pageTitle="Sign In"

      <LoginForm />
    </Layout>
  )

```

### Section-based Layout

```tsx

  return (

      </Section>

        <AboutContent />
      </Section>

      <Section spacing="md" contained={false}>
        <FullWidthGallery />
      </Section>
    </Layout>
  )
}
```
